## Method 3 — Python (automated, headless)

Install dependencies:
```bash
pip install playwright
playwright install chromium
```

Run capture script:
```python
from playwright.sync_api import sync_playwright
from datetime import datetime

def capture(url="http://localhost:5050",
            out="outputs/reports/dashboard.png"):
    with sync_playwright() as p:
        browser = p.chromium.launch()
        page    = browser.new_page(viewport={"width": 1600, "height": 1200})
        page.goto(url, wait_until="networkidle")
        page.wait_for_timeout(3000)          # let charts render
        page.screenshot(path=out, full_page=True)
        browser.close()
        print(f"Saved → {out}")

if __name__ == "__main__":
    capture()
```

## Method 4 — Selenium (Python)

```python
from selenium import webdriver
from selenium.webdriver.chrome.options import Options

opts = Options()
opts.add_argument("--headless")
opts.add_argument("--window-size=1600,1200")
driver = webdriver.Chrome(options=opts)
driver.get("http://localhost:5050")
driver.implicitly_wait(5)
driver.save_screenshot("outputs/reports/dashboard.png")
driver.quit()
```

## Automated daily screenshot (cron)

```cron
# Every day at 07:00 local time
0 7 * * * /usr/bin/python3 /path/to/project/dashboard/capture.py
```

`dashboard/capture.py`:
```python
from playwright.sync_api import sync_playwright
from datetime import datetime
from pathlib import Path

Path("outputs/reports").mkdir(parents=True, exist_ok=True)
ts  = datetime.now().strftime("%Y%m%d_%H%M%S")
out = f"outputs/reports/dashboard_{ts}.png"

with sync_playwright() as p:
    br   = p.chromium.launch()
    page = br.new_page(viewport={"width": 1600, "height": 1200})
    page.goto("http://localhost:5050", wait_until="networkidle")
    page.wait_for_timeout(4000)
    page.screenshot(path=out, full_page=True)
    br.close()
print(f"Saved → {out}")
```
